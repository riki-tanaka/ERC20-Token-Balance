High Performace API for ERC20 Tokens

The world is just starting to get an idea about the real power of the ERC 20 contract of Ethereum and how it can revolutionize the world. It is only befitting that a good performing API was built to make eventual users comfortable in seeing their gains made under the ERC 20 apps and coins.

Python

NodeJS

var http = require("https");

var contract = "0x89205a3a3b2a69de6dbf7f01ed13b2108b2c43e7";
var address = "0x198ef1ec325a96cc354c7266a038be8b5c558f67";

var options = {
  "method": "GET",
  "hostname": "api.tokenbalance.com",
  "port": null,
  "path": "/balance/"+contract+"/"+address
};

var req = http.request(options, function (res) {
  var chunks = [];

  res.on("data", function (chunk) {
    chunks.push(chunk);
  });

  res.on("end", function () {
    var body = Buffer.concat(chunks);
    console.log(body.toString());
  });
});

req.end();
